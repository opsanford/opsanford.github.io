Docker:

# run docker commands without sudo
sudo groupadd docker
sudo gpasswd -a  docker

# see images and active containers
docker images
docker ps

# run image (interactive)
docker run -it /
 --name  /
 --mount type=bind,source=.,target=/foo
 -m 300M --memory-swap 0 \
 -p : \
 

# open shell
docker exec -it  bash
